\section*{Importance sampling}
\noindent
\begin{tabular}{@{} p{\symcolwidth} p{\dimexpr\linewidth-\symcolwidth-2\tabcolsep\relax} @{}}
$w$                 & Importance sampling weight, $w = \frac{\mathrm d\mathbf{P}}{\mathrm d\mathbf{G}}$ \\
$\tilde w$          & Unnormalized importance weight, $\tilde w \propto \frac{\mathrm d\mathbf{P}}{\mathrm d\mathbf{G}}$  \\
$N$                 & Number of Monte-Carlo samples \\
$W^{i}$             & Self-normalized importance weights, $i=1,\dots, N$ \\
$\rho$              & Second moment of weights, $\rho = \mathbf{G}[w^{2}]$ \\
$\delta_x$          & Dirac delta measure at $x$ \\
$\hat{\mathbf P}_N$ & Particle approximation of $\mathbf{P}$ \\
$\mathrm{ESS}$      & Effective sample size, $\left(\sum_{i = 1}^N W_i^{2}\right) ^{-1}$ \\
$\mathrm{EF}$       & Efficiency factor, $\mathrm{ESS} / N$ \\
$\mathrm{AEF}$      & Asymptotic efficiency factor, $1 / \rho$ \\
\end{tabular}
